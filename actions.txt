const ADD_TODO = "ADD_TODO";
function addTodo(todoText) {
    return {
        type: ADD_TODO,
        todoText: todoText
    };
}

const REMOVE_TODO = "REMOVE_TODO"
function removeTodo(id) {
    return {
        type: REMOVE_TODO,
        id: id
    }   
}

const EDIT_TODO_STATUS = "EDIT_TODO_STATUS" ;
function editTodoStatus(id, status) {         
    return {
        type: EDIT_TODO_STATUS,
        id,          
        status
    };

}

const START_EDITING_TODO_TEXT = "START_EDITING_TODO_TEXT" ;
function startEditingTodo(todoText) {
    return {
        type: START_EDITING_TODO_TEXT
        todoText:todoText
    };
}

const EDIT_TODO_TEXT = "EDIT_TODO_TEXT";
function editTodoText(id, todoText) {
    return {
      type: EDIT_TODO_TEXT
      id,
      todoText
    };
}

const STOP_EDITING_TODO = "STOP_EDITING_TODO;
function stopEditingTodo(id, todoText) {
    return {
        type: STOP_EDITING_TODO
        id,
        todoText 
    };
}
